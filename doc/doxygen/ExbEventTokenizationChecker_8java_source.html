74  @Override
75  public Report function(CorpusData cd, Boolean fix) // check whether there's any illegal apostrophes '
76  throws SAXException, IOException, ParserConfigurationException, URISyntaxException, JDOMException, TransformerException, XPathExpressionException {
77  Report stats = new Report(); // create a new report
78  doc = TypeConverter.String2JdomDocument(cd.toSaveableString()); // read the file as a doc
79  Pattern excludePattern = Pattern.compile("^\\(\\(.*\\)\\)\\s*$"); // ignore events which contain information in double parentheses
80  Pattern whitespacePattern = Pattern.compile("\\S\\s\\S");
81  String xpathSpeakers = "//tier[@type='" + tokensTierName + "']//@speaker";
82  XPath speakers = XPath.newInstance(xpathSpeakers);
83  List allSpeakers = speakers.selectNodes(doc);
84  CorpusIO cio = new CorpusIO();
85  for (int sp = 0; sp < allSpeakers.size(); sp++) {
86  Object ob = allSpeakers.get(sp);
87  if (ob instanceof Attribute) {
88  Attribute attr = (Attribute) ob;
89  String speakerName = attr.getValue();
90  String xpathContext = "//tier[@type='" + tokensTierName + "'][@speaker='" + speakerName + "']/event";
91  XPath context = XPath.newInstance(xpathContext);
92  List allContextInstances = context.selectNodes(doc);
93  if (!allContextInstances.isEmpty()) {
94  for (int i = 0; i < allContextInstances.size(); i++) {
95  Object o = allContextInstances.get(i);
96  if (o instanceof Element) {
97  Element e = (Element) o;
98  String s = e.getText();
99  String st = e.getAttributeValue("start");
100  String end = e.getAttributeValue("end");
101  if (st == null || end == null) {
102  String message = "Timestamp missing in an event: " + escapeHtml4(s);
103  exmaError.addError(function, cd.getURL().getFile(), tokensTierName, st, false, message);
104  stats.addCritical(function, cd, message);
105  continue;
106  }
107  if (excludePattern.matcher(s).find()) {
108  continue; //ignore excluded events
109  }
110  if (whitespacePattern.matcher(s).find()) { // if file contains the RegEx then issue warning
111  badTokens = true;
112  String message = "Tokenization is not complete in an event: " + escapeHtml4(s);
113  exmaError.addError(function, cd.getURL().getFile(), tokensTierName, st, false, message);
114  stats.addCritical(function, cd, message);
115  continue;
116  }
117  for (int j = 0; j < lsTiersToCheck.size(); j++) {
118  String curTierName = lsTiersToCheck.get(j);
119  String xpathComp = "//tier[@category='" + curTierName + "'][@speaker='" + speakerName + "']/event[@start='" + st + "']";
120  XPath compl = XPath.newInstance(xpathComp);
121  List complInstances = compl.selectNodes(doc);
122  if (!complInstances.isEmpty()) {
123  Object oo = complInstances.get(0);
124  if (oo instanceof Element) {
125  Element ee = (Element) oo;
126  String end_a = ee.getAttributeValue("end");
127  if (end_a == null) {
128  String message = "Timestamp missing in an event: " + escapeHtml4(s);
129  exmaError.addError(function, cd.getURL().getFile(), curTierName, st, false, message);
130  stats.addCritical(function, cd, message);
131  continue;
132  }
133  if (!end.equals(end_a)) {
134  incompleteAnnotation = true;
135  String message = "The event at " + st + " for the speaker" + speakerName + " seems to be missing annotation in the tier " + curTierName;
136  exmaError.addError(function, cd.getURL().getFile(), curTierName, st, false, message);
137  stats.addCritical(function, cd, message);
138  }
139  }
140  }
141  else {
142  incompleteAnnotation = true;
143  String message = "The event at " + st + " for the speaker" + speakerName + " seems to be missing annotation in the tier " + curTierName;
144  exmaError.addError(function, cd.getURL().getFile(), curTierName, st, false, message);
145  stats.addCritical(function, cd, message);
146  }
147  }
148  } else {
149  stats.addCorrect(function, cd, "CorpusData file does not contain any event");
150  }
151  }
152  }
153  }
154  }
155  if (!badTokens) {
156  stats.addCorrect(function, cd, "Tokenization OK");
157  }
158  if (!incompleteAnnotation) {
159  stats.addCorrect(function, cd, "Annotation is complete");
160  }
161  return stats; // return the report with warnings
162  }
