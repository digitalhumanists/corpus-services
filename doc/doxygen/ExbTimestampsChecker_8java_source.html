58  CorpusIO cio = new CorpusIO();

60  Object o = allSegments.get(i);
61  if (o instanceof Element) {
62  Element e = (Element) o;
63  String start = e.getAttributeValue("s");
64  String end = e.getAttributeValue("e");
65  String xpathStart = "//tli[@id='" + start + "']";
66  XPath timelineStart = XPath.newInstance(xpathStart);

68  Object sTli = tliStart.get(0);
69  if (sTli instanceof Element) {
70  Element el = (Element) sTli;
71  String id = el.getAttributeValue("id");
72  String time = el.getAttributeValue("time");
73  if (time == null) {
74  missingTimestamp = true;
75  String message = "Missing timestamp at the start of the segment chain at " + id;
76  exmaError.addError(function, cd.getURL().getFile(), "", id, false, message);
77  stats.addWarning(function, cd, message);
78  }
79  }

81  XPath timelineEnd = XPath.newInstance(xpathEnd);
