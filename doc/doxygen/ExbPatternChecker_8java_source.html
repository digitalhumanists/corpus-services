72  public Report exceptionalCheck(File f)
73  throws SAXException, IOException, ParserConfigurationException {
74  // XXX: get conventions from settings somehow
75  List<Pattern> correctPatterns = new ArrayList<Pattern>();
76  for (String convention : conventions) {
77  correctPatterns.add(Pattern.compile(convention));
78  }
79  List<Pattern> errorPatterns = new ArrayList<Pattern>();
80  for (String problem : problems) {
81  errorPatterns.add(Pattern.compile(problem));
82  }
83  DocumentBuilderFactory dbf = DocumentBuilderFactory.newInstance();
84  DocumentBuilder db = dbf.newDocumentBuilder();
85  Document doc = db.parse(f);
86  NodeList events = doc.getElementsByTagName("event");
87  Report stats = new Report();
88  for (int i = 0; i < events.getLength(); i++) {
89  Element event = (Element)events.item(i);
90  NodeList eventTexts = event.getChildNodes();
91  for (int j = 0; j < eventTexts.getLength(); j++) {
92  Node maybeText = eventTexts.item(j);
93  if (maybeText.getNodeType() != Node.TEXT_NODE) {
94  if (maybeText.getNodeType() == Node.ELEMENT_NODE &&
95  maybeText.getNodeName().equals("ud-information")) {
96  // XXX: ud-information is weird I'll just skip it...
97  continue;
98  }
99  System.err.println("This is not a text node: " +
100  maybeText);
101  continue;
102  }
103  Text eventText = (Text) maybeText;
104  String text = eventText.getWholeText();
105  int k = 0;
106  for (Pattern pattern : correctPatterns) {
107  Matcher matcher = pattern.matcher(text);
108  if (!matcher.matches()) {
109  stats.addCritical(EXB_PATTERNS,
110  "Text: " + text + " does not fit to the " +
111  "conventions given.", "Expression was: " +
112  conventions.get(k));
113  }
114  }
115  k = 0;
116  for (Pattern pattern : errorPatterns) {
117  Matcher matcher = pattern.matcher(text);
118  if (matcher.matches()) {
119  stats.addCritical(EXB_PATTERNS,
120  "Text: " + text + " does not fit to the " +
121  "conventions given.", "Expression was: " +
122  errorPatterns.get(k));
123  }
124  }
125  }
126  }
127  return stats;
