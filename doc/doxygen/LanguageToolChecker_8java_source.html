95  for (int k = 0; k < tiers.getLength(); k++) {

97  if (!tier.getAttribute("category").equals(tierToCheck)) {

101  for (int i = 0; i < events.getLength(); i++) {
102  Element event = (Element)events.item(i);
103  NodeList eventTexts = event.getChildNodes();
104  for (int j = 0; j < eventTexts.getLength(); j++) {
105  Node maybeText = eventTexts.item(j);
106  if (maybeText.getNodeType() != Node.TEXT_NODE) {
107  if (maybeText.getNodeType() == Node.ELEMENT_NODE &&
108  maybeText.getNodeName().equals("ud-information")) {
109  // XXX: ud-information is weird I'll just skip it...
110  continue;
111  }
112  System.err.println("This is not a text node: " +
113  maybeText);
114  continue;
115  }
116  Text eventText = (Text) maybeText;
117  String text = eventText.getWholeText();
118  List<RuleMatch> matches = langTool.check(text);
119  for (RuleMatch match : matches) {
120  stats.addWarning(LANGUAGETOOL,
121  "Potential error at characters " +
122  match.getFromPos() + "-" + match.getToPos() + ": " +
123  match.getMessage() + ": \"" +

125  match.getToPos()) + "\" ",
126  "Suggested correction(s): " +

128  }
129  }
130  }
